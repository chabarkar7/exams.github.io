using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace Netprac5
{
    public partial class WebForm1 : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            Response.Cookies["computer"].Expires = DateTime.Now.AddDays(-1);
        }

        protected void Button1_Click(object sender, EventArgs e)
        {
            Label.Text = "";
            //Adding values
            if (apple.Checked)
                Response.Cookies["computer"]["apple"] = "Apple";
            if (dell.Checked)
                Response.Cookies["computer"]["dell"] = "Dell";
            if (lenevo.Checked)
                Response.Cookies["computer"]["lenevo"] = "Lenevo";
            if (acer.Checked)
                Response.Cookies["computer"]["acer"] = "Acer";
            if (sony.Checked)
                Response.Cookies["computer"]["sony"] = "Sony";
            if (wipro.Checked)
                Response.Cookies["computer"]["wipro"] = "Wipro";
            //Fetching values
            if (Request.Cookies["computer"].Values.ToString() != null)
            {
                if (Request.Cookies["computer"]["apple"] != null)
                    Label.Text += Request.Cookies["computer"]["apple"] + " ";
                if (Request.Cookies["computer"]["dell"] != null)
                    Label.Text += Request.Cookies["computer"]["dell"] + " ";
                if (Request.Cookies["computer"]["lenevo"] != null)
                    Label.Text += Request.Cookies["computer"]["lenevo"] + " ";
                if (Request.Cookies["computer"]["acer"] != null)
                    Label.Text += Request.Cookies["computer"]["acer"] + " ";
                if (Request.Cookies["computer"]["sony"] != null)
                    Label.Text += Request.Cookies["computer"]["sony"] + " ";
                if (Request.Cookies["computer"]["wipro"] != null)
                    Label.Text += Request.Cookies["computer"]["wipro"] + " ";
            }
            else Label.Text = "Please select your choice";
            Response.Cookies["computer"].Expires = DateTime.Now.AddDays(-1);


        }
    }
}

                Application.UnLock();

            void Session_Start(object sender, EventArgs e)
            {
                int tusers, ousers;
                Application.Lock();
                tusers = int.Parse(Application["tusers"].ToString()) + 1;
                ousers = int.Parse(Application["ousers"].ToString()) + 1;
                Application["tusers"] = tusers;
                Application["ousers"] = ousers;
                Application.UnLock();
            }

            void Session_End(object sender, EventArgs e)
            {
                int ousers;
                Application.Lock();
                ousers = int.Parse(Application["ousers"].ToString()) - 1;
                Application["ousers"] = ousers;
                Application.UnLock();
